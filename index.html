Classes

Syntactical sugar over JavaScript's existing prototype-based inheritance

Classes provide a much simpler and clearer syntax to create objects and deal with inheritance

The bodies of class declarations and class expressions are executed in strict mode

Classes

Before


function Vehicle (name, type) {
    this.name = name;
    this.type = type;
};
 
Vehicle.prototype.getName = function getName () {
    return this.name;
};
 
Vehicle.prototype.getType = function getType () {
    return this.type;
};

Classes

Now


class Vehicle {
    constructor(name, type) {
        this.name = name;
        this.type = type;
    }

    getName() {
        return this.name;
    }

    getType() {
        return this.type;
    }
}

Classes

Simplier inheritance


class Car extends Vehicle {
    constructor(name) {
        super(name, 'car');
    }

    getName() {
        return 'It is a car: ' + super.getName();
    }
}

Classes

get / set / static


class Car {
    constructor(name) {
        this._name = name;
    } 

    set name(name) {
        this._name = name;
    }

    get name() {
        return this._name;
    }

    static create(name, type) {
        return new Car(name, type);
    }
}
Car.create("Ford", "car"); // return new Car

Classes

Class expression


let Car = class {
    constructor(name) {
        this._name = name;
    }
};

Arrow functions

New shorter syntax for function expression

Before


var test = function (value) {
    return value * 2;
};

Now


let test = (value) => {
    return value * 2;
};

Arrow functions

IIFE with Arrow Function


(() => {
    // some code
}());

Even shorter syntax for functions with only return statement


let test = (value) => value * 2;

Arrow functions - Automatic binding

Before


function Person() {
    var that = this;
    that.age = 0;

    setInterval(function() {
        that.age++;
    }, 1000);
}

Now


function Person(){
    this.age = 0;

    setInterval(() => {
        this.age++;
    }, 1000);
}

Modules

Now we can write modular JS code natively!

ES6 Modules are in strict mode by default (bye bye "use strict")

Modules


export const sqrt = Math.sqrt;
export function square(x) {
    return x * x;
}
export  function diag(x, y) {
    return sqrt(square(x) + square(y));
}

export default function test() { return "test"; }

// ------ 
import { square, diag } from 'lib';
import test from 'lib';
console.log(square(11));   // 121
console.log(diag(4, 3));   // 5

    Symbols, Maps, Sets

    New types of objects

    Symbol

    They are tokens that serve as unique IDs

    
let symbol1 = Symbol();
let symbol2 = Symbol('symbol2');

    Every symbol returned by Symbol() is unique

    
symbol1 === symbol2; // false

    Map

    The Map object holds key-value pairs

    
let map = new Map();

map.set('foo', 123);
map.get('foo');  // 123

map.has('foo'); // true
map.delete('foo');
map.has('foo');  // false

    Map

    Setting items in constructor

    
let map = new Map([
    [ 1, 'one' ],
    [ 2, 'two' ],
    [ 3, 'three' ]
]);

    WeakMap

    Similiar to standard Map object but every key has to be an object

    WeakMap doesn’t prevent its keys from being garbage-collected

    
var map = new WeakMap();
map.set(1, 2);
// TypeError: 1 is not an object!

    Set

    It’s a collection for unique values

    
let set = new Set();
set.add(1);
set.add('1');
set.add({ key: 'value' });
console.log(set); // Set {1, '1', Object {key: 'value'}}

    Set doesn’t allow to add duplicates!

    WeakSet

    Is a Set that doesn’t prevent its values from being garbage-collected

    
let set = new WeakSet();
set.add(1);
set.add('1');
set.add({ key: 'value' });
console.log(set); // WeakSet {1, '1', Object {key: 'value'}}

    WeakSet doesn’t allow to add duplicates!

    Async Functions (ES8)

    The hottest thing!

    • It is a new way to write asynchronous code
    • Async function is built on top of promises
    • Async function is non blocking
    • Code is prettier, because looks like synchronous

    Async Functions (ES8)

    
const makeRequest = async () => {
    console.log(await getJSON());
    return "done";
};

    Error handling

    
const makeRequest = async () => {
    try {
        console.log(await getJSON());
    } catch (err) {
        console.error(err);
    }
};
